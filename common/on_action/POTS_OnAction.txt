on_uchiha_birth = {
	effect = {
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_uchiha = yes
					}
					scope:mother = {
						is_uchiha = yes
					}
				}
			}
			add_trait = uchiha
		}
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_uchiha = yes
					}
					scope:mother = {
						is_uchiha = no
					}
				}
			}
			add_trait = uchiha
		}
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_uchiha = no
					}
					scope:mother = {
						is_uchiha = yes
						patrilinear_marriage = no
					}
				}
			}
			add_trait = uchiha
		}
	}
}

on_forbidden_birth = {
	effect = {
		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_uchiha = yes
						}
						scope:mother = {
							is_hyuga = yes
						}
					}
					AND = {
						scope:real_father = {
							is_hyuga = yes
						}
						scope:mother = {
							is_uchiha = yes
						}
					}
				}
				NOT = { is_hyuga = yes
						is_uchiha = yes
				}
			}
			random_list = {
				30 = {
					add_trait = forbiddengene
					add_trait = forbiddeneyes
				}
				35 = {
					add_trait = uchiha
				}
				35 = {
					add_trait = hyuga
					random_list = {
						95 = {
							add_trait = byakugan
						}
						5 = {
							add_trait = jogan
						}
					}
				}
			}
		}
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_forbidden = yes
					}
					scope:mother = {
						is_forbidden = yes
					}
				}
			}
			add_trait = forbiddengene
			add_trait = forbiddeneyes
		}
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_forbidden = yes
					}
					scope:mother = {
						is_forbidden = no
					}
				}
			}
			add_trait = forbiddengene
			add_trait = forbiddeneyes
		}
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_forbidden = no
					}
					scope:mother = {
						is_forbidden = yes
						patrilinear_marriage = no
					}
				}
			}
			add_trait = forbiddengene
			add_trait = forbiddeneyes
		}
	}
}
