on_birthday = {
    on_actions = {
        sharingan_birthday_one
        sharingan_birthday_two
        sharingan_birthday_three
    }
}

sharingan_birthday_one = {
    trigger = {
        AND = {
            age = 20
            has_trait = uchiha
            is_ai = yes
        }
    }

    events = {
        sharingan_level.1
    }
}

sharingan_birthday_two = {
    trigger = {
        AND = {
            age = 35
            has_trait = sharingan_tomoe_1
            is_ai = yes
        }
    }

    events = {
        sharingan_level.2
    }
}

sharingan_birthday_three = {
    trigger = {
        AND = {
            age = 60
            has_trait = sharingan_tomoe_2
            is_ai = yes
        }
    }

    events = {
        sharingan_level.3
    }
}
